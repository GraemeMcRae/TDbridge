# Force polling mode instead of webhook mode (true/false; default is false on VPS Linux, default is true on Windows Gitbash or Windows WSL2).

# SQLite message store file (stores TG<->DC message ID mappings for reply routing)
TEST_SQLITE_DB_FILE=TDbridge_test.db
# Log file

# What to do in Telegram when a Discord message is deleted: "delete" | "<message to user>" | "ignore"

# Message to post on Telegram if deletion of the Telegram message fails, or empty string for no message

# Message to issue when a Telegram message in an Active group can't be routed to any Discord channel/user/role, or empty string to issue no message

# Message to issue when a Telegram message was routed to an inactive Discord user/role, or empty string to issue no message
TEST_ROUTED_INACTIVE_TTOD_ERRMSG="! Message routed to inactive Discord user."
# Message to issue when a Discord message on an Active channel can't be routed to any Telegram group, or empty string to issue no message
TEST_UNROUTABLE_DTOT_ERRMSG="! Unable to route this message to Telegram."
# How to bridge reactions from Telegram to Discord: "react" | "reply" | "both" | "neither"
TEST_REACTIONS_TTOD=both
# How to bridge reactions from Discord to Telegram: "react" | "reply" | "both" | "neither"
TEST_REACTIONS_DTOT=both
# On Telegram, reply to a message you want to delete with a message matching this regex (using re.fullmatch)
# Use \s* to match whitespace, e.g. "(?i)delete\s*". Use the empty string to disable this feature entirely.

# Error message posted on Telegram when a delete command fails. Empty string (the default) means no error message is posted.

# TLS certificate for the Telegram webhook HTTPS server.
# Both test and prod share the same certificate (no TEST_/PROD_ prefix)
# because both instances run on the same server with the same domain name.
# The Let's Encrypt cert is obtained via: sudo certbot certonly --manual --preferred-challenges dns -d hcf.squadrontrucking.com
# Certificate expires every 90 days. Renewal procedure is documented in TDbridge_Project_Structure.md.
# The graeme user must have read access to these files:
#   sudo chgrp -R graeme /etc/letsencrypt/live /etc/letsencrypt/archive
#   sudo chmod 750 /etc/letsencrypt/live /etc/letsencrypt/archive
#   sudo chmod 750 /etc/letsencrypt/live/hcf.squadrontrucking.com /etc/letsencrypt/archive/hcf.squadrontrucking.com
TLS_CERT_FILE=/etc/letsencrypt/live/hcf.squadrontrucking.com/fullchain.pem
